#version 330 core
out vec4 fragColor;

uniform vec3 iResolution;   // viewport resolution (in pixels)
uniform float iTime;        // shader playback time (in seconds)
uniform vec4 iMouse;        // xy: current (if MLB down), zw: click

vec3 sdfCircle(vec2 uv, float r) {
    float x = uv.x;
    float y = uv.y;

    float d = length(vec2(x, y)) - r;

    return d > 0. ? vec3(1.) : vec3(0., 0., 1.);
}

void main()
{
    // Normalized pixel coordinates (from 0 to 1)
    vec2 uv = gl_FragCoord.xy / iResolution.xy;

    vec2 id = floor(uv*30.0)/60.0;
    uv = fract(uv * 30.0 );
    
	uv -= 0.5; // <-0.5, 0.5>
    uv.x *= iResolution.x/iResolution.y; // fix aspect ratio
    // Call this function on each pixel to check if the coordinate lies inside or outside of the circle
    vec3 col = sdfCircle(uv, id.x); 

    fragColor = vec4(col, 1.0);
}
