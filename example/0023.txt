#version 330 core
out vec4 fragColor;

uniform vec3 iResolution;   // viewport resolution (in pixels)
uniform float iTime;        // shader playback time (in seconds)
uniform vec4 iMouse;        // xy: current (if MLB down), zw: click

void main() {
    vec2 uv = gl_FragCoord.xy/iResolution.xy; // <0, 1>

    uv = fract(uv * 30.0);//cut to 30 segments

    //float interpolatedValue = mix(0., 1., uv.x);//columns
    float interpolatedValue = mix(0., 1., uv.y);//rows
    vec3 col = vec3(interpolatedValue);//black to white gradient.

    // Output to screen
    fragColor = vec4(col,1.0);
}
